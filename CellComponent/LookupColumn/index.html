}, {
    name: "Gender",
    fieldName: "Gender",
    type: "data",
    width: "100",
    sortable: false,
    lookupDisplay: true,
    editor: {
        type: "dropDown",
        dropDownCount: 2,
        domainOnly: true,
        values: ["Male","Female"],
        labels: ["<Male>","<Female>"]
    },
    styles: {
        textAlignment: "center"
    },
    header: {
        text: "Gender",
        styles: {

        }
    }

	...
}];

grid.setColumns(columns);
